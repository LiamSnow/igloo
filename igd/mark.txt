IGLOO VALUE -> APPLICABLE ELEMENTS

Integer -> NumberPicker, Slider, DurationPicker
Real -> NumberPicker, Slider
Text -> Text, TextInput, Icon, TextSelect
Boolean -> Switch, Checkbox, ToggleButton
Color -> ColorPicker
Date -> DatePicker
Time -> TimePicker
Enum -> EnumSelect
Text + TextList -> CustomSelect

*Real + *Integer -> Chart 
? -> Table

IntegerList | RealList | TextList | BooleanList | ColorList | DateList | TimeList -> ForEach

(Marker/Components without an Igloo Value) -> Button
